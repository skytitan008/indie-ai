```bash

python3 cli.py run multi-agent
python3 cli.py run task
python3 cli.py run long-term
python3 cli.py run integration

# 查看状态
python3 cli.py status

# 查看统计
python3 cli.py stats

# 对比实验
python3 cli.py compare learning --rounds 5
python3 cli.py compare decision --rounds 5

# 清理数据
python3 cli.py clean

# 启动服务

# 其他
python3 cli.py version
python3 cli.py --help
```
